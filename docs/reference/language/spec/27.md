## 15. 종합 예시 (v0)

```parus
struct Vec2 {
  x: i32;
  y: i32;
}

proto Summable {
  def sum(self) -> i32 {
    return 0i32;
  }
}

class Pair: Summable {
  x: i32;
  y: i32;
  init() = default;
  deinit() = default;

  def sum(self) -> i32 {
    return self.x + self.y;
  }
}

acts for Vec2 {
  def add(self) -> i32 {
    return self.x + self.y;
  }
}

actor Counter {
  draft {
    value: i32;
  }

  init(seed: i32) {
    draft.value = seed;
  }

  def sub get() -> i32 {
    recast;
    return draft.value;
  }

  def pub add(delta: i32) -> i32 {
    draft.value = draft.value + delta;
    commit;
    return draft.value;
  }
}

def main() -> i32 {
  set v = Vec2 { x: 1i32, y: 2i32 };
  set p = Pair();
  set c = spawn Counter(seed: 10i32);

  set a = v.add();
  set b = p.sum();
  set x = c.get();
  set y = c.add(delta: 3i32);

  return a + b + x + y;
}
```
