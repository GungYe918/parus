field Vec2 {

field<u32, i32> OnlyInts {

  def draw(self, ctx: &mut RenderCtx) -> void;
}

acts for u32 {
  operator(+)(self, rhs: u32) -> u32 {
    return __intrin_u32_add(a: self, b: rhs);
  }

  operator(==) -> u32 {
    return __intrin_u32_eq(a: self, b: rhs);
  }
}

actor Counter {
  def sub get() -> u32 {
    // 필요하면 관찰 뷰를 재설정
    recast;
    return draft.count;
  }

  def pub inc() -> void {
    draft.count += 1u32;
    commit;
  }
}

def f(a: int) -> int { return a; }

def f(a: int, b: int) -> int { return a + b; }

def main() -> void {
  let xs: u32[3] = [1u32, 2u32, 3u32];

  set mut s = 0u32;
  loop(iter: v in xs) {
    s = s + v;
  }

  // loop(for: ...) 예시 (명시 이름 버전)

  loop(for: i in 1..:5) {
    t = t + i;
  }

  set mut x = 10;
  set y = x++;
  set z = ++x;

  }

  set b = f(a: 1, b: 2);
